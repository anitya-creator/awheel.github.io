const links = document.querySelectorAll("a"); const transition = document.querySelector(".page-transition"); links.forEach(link=>{ link.addEventListener("click",function(e){ const target = this.href; if(target.includes("#")) return; e.preventDefault(); transition.classList.add("active"); setTimeout(()=>{ window.location = target; },800); }); }); window.onload = () =>{ document.querySelector(".page-transition").classList.remove("active"); }
